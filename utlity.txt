package base;

import java.time.Duration;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.support.ui.ExpectedConditions;
import org.openqa.selenium.support.ui.WebDriverWait;
import org.testng.annotations.AfterMethod;

public class Utility {

	protected WebDriver driver;
	WebDriverWait wait;

	// Constructor
	public Utility() {
		
	}	
	
	public Utility(WebDriver driver) {
		this.driver = driver;
		this.wait = new WebDriverWait(driver, Duration.ofSeconds(10));
	}

	// Locators
	private By popuphandler = By.xpath("//div[@id='omegaPromoPopupOne']");
	private By close = By.xpath("(//button[@aria-label='Close'])[8]");

	// Popup Handler Function

	public void popupHandler() {
		try {
			wait.until(ExpectedConditions.visibilityOfElementLocated(popuphandler));
			WebElement button = wait.until(ExpectedConditions.elementToBeClickable(close));
			button.click();
			System.out.println("Popup closed");
		} catch (Exception e) {
			System.out.println("Popup is not present");
		}
	}

	public void setUp(String url) {
		System.setProperty("webdriver.chrome.driver",
				"F:\\Anil Patil\\Kanif\\Drivers\\chromedriver_win32\\chromedriver.exe");
		driver = new ChromeDriver();
		driver.manage().window().maximize();
		driver.get(url);

	}

	@AfterMethod
	public void tearDown() throws InterruptedException {
		Thread.sleep(3000);
		driver.quit();
	}

}
